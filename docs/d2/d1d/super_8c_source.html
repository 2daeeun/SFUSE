29int sb_load(int fd, struct sfuse_super *sb) {
30 ssize_t ret;
31
32 // 슈퍼블록을 디스크(SFUSE_SUPERBLOCK_OFFSET)에서 읽어 메모리(sb)에 저장한다.
33 ret = disk_read(fd, sb, sizeof(*sb), SFUSE_SUPERBLOCK_OFFSET);
34
35 // disk_read가 실패하면 반환된 음수 오류 값을 그대로 반환한다.
36 if (ret < 0)
37 return ret;
38
39 // 요청한 슈퍼블록 크기만큼 정확히 읽었는지 확인한다.
40 if ((size_t)ret != sizeof(*sb))
41 return -EIO; // 읽은 데이터 크기가 슈퍼블록 크기와 다름 (입출력 오류)
42
43 // 슈퍼블록의 매직 넘버를 검사하여 슈퍼블록 유효성 검사를 수행한다.
44 if (sb->magic != SFUSE_MAGIC)
45 return -EINVAL; // 잘못된 매직 넘버
46
47 // 성공적으로 슈퍼블록을 읽고 유효성을 확인했으므로 0 반환.
48 return 0;
49}

63int sb_sync(int fd, const struct sfuse_super *sb) {

65
66 // 슈퍼블록 내용을 디스크(SFUSE_SUPERBLOCK_OFFSET)에 기록
67 ret = disk_write(fd, sb, sizeof(*sb), SFUSE_SUPERBLOCK_OFFSET);
68
69 // 쓰기 실패 시, disk_write 함수가 반환한 음수 오류 코드를 반환
70 if (ret < 0)
71 return ret;
72
73 // 요청한 슈퍼블록 크기만큼 정확히 기록되었는지 확인
74 if ((size_t)ret != sizeof(*sb))
75 return -EIO;
76
77 // 정상적으로 슈퍼블록이 디스크에 저장됨
78 return 0;
