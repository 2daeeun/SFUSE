75int fs_initialize(int backing_fd) {
76 // 현재 사용 중인 파일 시스템 컨텍스트 획득
77 struct sfuse_fs *fs = get_fs_context();
78
79 // 얻은 블록 장치 파일 디스크립터 저장
80 fs->backing_fd = backing_fd;
81
82 /*
83 블록 장치(backing_fd)의 크기를 바이트 단위로 얻는다.
84 파일의 크기를 얻으려면 파일의 가장 끝으로 파일 오프셋(offset)을 이동시킨다.
85 이때 사용되는 함수가 바로 `lseek`이다.
86
87 lseek 함수 원형:
88 off_t lseek(int fd, off_t offset, int whence);
89
90 인자 설명:
91 - fd: 오프셋을 이동시킬 파일 디스크립터.
92 - offset: 이동할 오프셋의 크기.
93 - whence: 오프셋 이동 기준을 설정하는 플래그.
94 - SEEK_SET: 파일의 시작 지점으로부터 offset 바이트 이동.
95 - SEEK_CUR: 현재 위치에서 offset 바이트 이동.
96 - SEEK_END: 파일 끝에서부터 offset 바이트 이동.
97
98 여기서는 파일의 크기를 알고자 하므로,
99 SEEK_END를 이용해 파일의 끝으로 이동(offset은 0)을 지정한다.
100 그러면 lseek는 파일의 총 크기(바이트 단위)를 반환한다.
101 */
102 off_t device_size = lseek(backing_fd, 0, SEEK_END);
103
104 /*
105 장치 크기(device_size)는 바이트 단위이기 때문에,
106 이를 SFUSE 파일 시스템이 사용하는 블록 크기(SFUSE_BLOCK_SIZE)로 나누어
107 블록 장치가 가진 전체 블록의 수를 계산한다.
108
109 예시로, 만약 장치 크기가 8192바이트이고 블록 크기가 1024바이트라면,
110 총 블록 수(total_blocks)는 8192 / 1024 = 8이 된다.
111
112 total_blocks 값은 파일 시스템의 메타데이터와 데이터 블록을 배치할 때
113 사용된다.
114 */
115 uint32_t total_blocks = device_size / SFUSE_BLOCK_SIZE;
116
117 // 슈퍼블록을 디스크에서 로드(load) 시도.
118 // sb_load()가 성공하면 이미 파일 시스템이 포맷되어 있다는 뜻이다.
119 // sb_load()는 디스크에서 슈퍼블록 데이터를 읽어 fs->sb 구조체에 저장한다.
120 // 성공 시 0을 반환하며, 실패 시 음수를 반환한다.
121 //
122 // sb_load()가 실패하면 (즉, 음수 반환 시),
123 // 이는 슈퍼블록이 존재하지 않거나 손상된 상태를 나타내므로,
124 // 새로운 슈퍼블록을 포맷하고 초기화하는 과정을 수행해야 한다.
125 if (sb_load(backing_fd, &fs->sb) < 0) {
126
127 // 슈퍼블록 포맷: 슈퍼블록을 깨끗한 초기 상태로 설정.
128 // 이 함수는 슈퍼블록의 모든 값을 기본값 또는 초기 상태로 설정한다.
129 sb_format(&fs->sb);
130
131 // 슈퍼블록에 파일 시스템의 전체 블록 개수(total_blocks)를 설정한다.
132 // 이 값은 파일 시스템이 관리할 수 있는 블록의 최대 개수를 나타낸다.
133 fs->sb.blocks_count = total_blocks;
134
135 // inode 개수 설정: 파일 시스템이 관리할 수 있는 inode의 총 개수를 설정.
136 // inode는 파일이나 디렉터리 같은 객체의 메타데이터를 저장하는 단위이다.
137 fs->sb.inodes_count = SFUSE_INODES_COUNT;
138
139 // --- 메타데이터 영역의 크기 계산 시작 ---
140
141 // 1. 블록 비트맵(block bitmap)의 크기를 블록 단위로 계산
142 //
143 // 블록 비트맵은 각 블록이 사용 중인지 여부를 비트 단위로 관리한다.
144 // 하나의 블록을 1비트(bit)로 관리하므로,
145 // 전체 블록 수를 8로 나누면 필요한 바이트(byte) 수가 된다.
146 //
147 // 이 바이트 수를 블록 단위(SFUSE_BLOCK_SIZE)로 변환하기 위해 다음 수식을
148 // 사용:
149 //
150 // (fs->sb.blocks_count / 8 + SFUSE_BLOCK_SIZE - 1) / SFUSE_BLOCK_SIZE
151 //
152 // 이 수식은 '올림(ceil)'의 효과를 내는 표현으로,
153 // 정확히 나누어 떨어지지 않을 경우 필요한 블록을 추가로 확보하기 위한
154 // 계산법이다.
155 uint32_t block_bitmap_blocks =
156 (fs->sb.blocks_count / 8 + SFUSE_BLOCK_SIZE - 1) / SFUSE_BLOCK_SIZE;
157
158 // 2. inode 비트맵(inode bitmap)의 크기를 블록 단위로 계산
159 //
160 // inode 비트맵은 inode들이 사용 중인지 여부를 관리하는 비트맵이다.
161 // 전체 inode 개수를 8로 나누면 바이트 수가 되며,
162 // 이것을 블록 단위로 변환하는 계산법은 블록 비트맵 계산법과 동일하다.
163 uint32_t inode_bitmap_blocks =
164 (fs->sb.inodes_count / 8 + SFUSE_BLOCK_SIZE - 1) / SFUSE_BLOCK_SIZE;
165
166 // 3. inode 테이블(inode table)의 크기를 블록 단위로 계산
167 //
168 // inode 테이블은 실제 inode 구조체들을 저장하는 영역이다.
169 // 각 inode는 'struct sfuse_inode'의 크기만큼 메모리를 차지하므로,
170 // 전체 inode 개수에 inode 크기를 곱하면 inode 테이블의 총 크기가 바이트
171 // 단위로 나온다.
172 //
173 // 이것을 다시 블록 크기(SFUSE_BLOCK_SIZE)로 나누어 몇 개의 블록이 필요한지
174 // 계산한다. 여기서도 블록이 정수로 딱 떨어지지 않을 수 있으므로, 올림
175 // 효과를 내도록 처리한다.
176 uint32_t inode_table_blocks =
177 (fs->sb.inodes_count * sizeof(struct sfuse_inode) + SFUSE_BLOCK_SIZE -
178 1) /
179 SFUSE_BLOCK_SIZE;
180
181 // 슈퍼블록에서 관리하는 각 영역의 위치(블록 번호)를 설정한다.
182 // SFUSE 파일 시스템의 디스크 레이아웃은 일반적으로 다음과 같다:
183 //
184 // 파일 시스템 블록 구조:
185 // +--------------------------+
186 // | 블록 0 (슈퍼블록) |
187 // +--------------------------+
188 // | 블록 1 (예비 블록) |
189 // +--------------------------+
190 // | 블록 2 ~ N |
191 // | (block bitmap) |
192 // | [블록 사용 여부 표시] |
193 // +--------------------------+
194 // | inode bitmap |
195 // | [inode 사용 여부 표시] |
196 // +--------------------------+
197 // | inode tables |
198 // | [파일 메타데이터 저장] |
199 // +--------------------------+
200 // | data block |
201 // | [실제 사용자 데이터 저장]|
202 // +--------------------------+
203 //
204 // block 비트맵의 시작 위치는 슈퍼블록(블록 0)과 예비 블록(블록 1) 뒤에 위치
205 fs->sb.block_bitmap_start = 2;
206
207 // inode 비트맵은 block 비트맵 다음에 위치
208 fs->sb.inode_bitmap_start = fs->sb.block_bitmap_start + block_bitmap_blocks;
209
210 // inode 테이블은 inode 비트맵 다음에 위치
211 fs->sb.inode_table_start = fs->sb.inode_bitmap_start + inode_bitmap_blocks;
212
213 // 실제 사용자 데이터 블록은 inode 테이블 다음에 위치
214 fs->sb.data_block_start = fs->sb.inode_table_start + inode_table_blocks;
215
216 // 데이터 블록 시작 위치 이후의 블록만 실제 데이터 저장을 위해 사용
217 // 가능하므로, 전체 블록 수(total_blocks)에서 데이터 블록 시작 위치를 빼면
218 // 여유 블록 수를 구할 수 있다.
219 fs->sb.free_blocks = total_blocks - fs->sb.data_block_start;
220
221 // inode 중 루트 디렉터리를 제외한 나머지 inode들이 사용 가능
222 fs->sb.free_inodes = fs->sb.inodes_count - 1;
223
224 // 위에서 설정한 슈퍼블록 데이터를 디스크에 즉시 동기화하여 저장
225 if (sb_sync(backing_fd, &fs->sb) < 0)
226 return -EIO; // 동기화 실패 시 입출력 에러 반환
227
228 // ---- 메모리에 블록 비트맵과 inode 비트맵을 관리할 공간을 확보 ----
229
230 // 비트맵 크기 계산:
231 // 비트맵은 비트 단위로 블록이나 inode의 사용 여부를 관리한다.
232 // 1 바이트(byte)는 8 비트(bit)를 가지므로, 블록과 inode의 개수를 각각 8로
233 // 나누면 비트맵이 필요한 메모리 크기가 바이트 단위로 나온다.
234 size_t bmap_bytes = fs->sb.blocks_count / 8;
235 size_t imap_bytes = fs->sb.inodes_count / 8;
236
237 // 위에서 계산한 크기만큼 메모리를 할당하고, 그 공간을 0으로 초기화한다.
238 // calloc(1, size)는 size 크기만큼 메모리를 할당한 후 모든 비트를 0으로
239 // 설정한다. 이렇게 하면 비트맵이 초기 상태에서는 모든 블록과 inode가 "사용
240 // 가능"(free)으로 표시된다.
241 fs->block_map = calloc(1, bmap_bytes);
242 fs->inode_map = calloc(1, imap_bytes);
243
244 // 메모리 할당 실패 시 메모리 부족 에러 반환
245 if (!fs->block_map || !fs->inode_map)
246 return -ENOMEM;
247
248 // 비트맵의 초기 상태(모든 블록 및 inode가 사용 가능 상태)를 디스크에 동기화
249 bitmap_sync(backing_fd, fs->sb.block_bitmap_start, fs->block_map,
250 bmap_bytes);
251 bitmap_sync(backing_fd, fs->sb.inode_bitmap_start, fs->inode_map,
252 imap_bytes);
253
254 // ---- 루트 디렉터리 inode 초기화 ----
255
256 // 루트 inode 번호 설정 (1번으로 정의됨)
257 uint32_t root = SFUSE_ROOT_INO;
258
259 // inode 비트맵에서 루트 inode를 사용 중으로 표시하는 코드
260 // 비트맵(bitmap)은 각 inode의 사용 여부를 관리하는 데이터 구조로,
261 // 1비트가 1개의 inode의 상태를 나타낸다.
262 // 비트값이 0이면 해당 inode는 사용 가능(free), 1이면 사용
263 // 중(allocated)이다.
264
265 // inode 비트맵은 바이트(byte) 배열 형태로 관리된다.
266 // 배열의 각 요소(바이트)마다 8개의 inode 상태를 표현할 수 있다.
267 // 특정 inode가 속한 바이트의 위치는 inode 번호를 8로 나눈 값(root / 8)이며,
268 // 바이트 내의 정확한 비트 위치는 inode 번호를 8로 나눈 나머지(root % 8)가
269 // 된다.
270
271 // 예시:
272 // 루트 inode(root)가 5라면, inode_map[0] (바이트의 첫 번째 요소)의 5번째
273 // 비트를 설정한다.
274
275 // 이 코드의 동작 방식:
276 // 1. (1 << (root % 8)): 1을 root % 8 만큼 왼쪽으로 시프트(shift)하여 원하는
277 // 비트 위치로 설정한다.
278 // 2. inode_map[root / 8] |= (...): OR 연산으로 해당 비트를 설정하여 inode를
279 // '사용 중'으로 표시한다.
280 //
281 // 비트 연산을 사용하는 이유는 매우 빠르고 공간을 절약하기 때문이다.
282 fs->inode_map[root / 8] |= (1 << (root % 8));
283
284 // 위에서 inode 비트맵에 루트 inode를 할당했기 때문에,
285 // 슈퍼블록에서 관리하는 사용 가능한(free) inode의 개수도 하나 감소시켜야
286 // 한다.
287 fs->sb.free_inodes--;
288
289 // 루트 inode 구조체 초기화 (초기화 전에 메모리 전체를 0으로 설정)
290 struct sfuse_inode root_inode;
291 memset(&root_inode, 0, sizeof(root_inode));
292
293 // 루트 inode 메타데이터 설정 (디렉터리, 접근 권한 0755, 소유자 UID/GID
294 // 설정)
295 fs_init_inode(&fs->sb, root, S_IFDIR | 0755, fuse_get_context()->uid,
296 fuse_get_context()->gid, &root_inode);
297
298 // ---- 루트 디렉터리의 데이터 블록 할당 및 초기화 ----
299
300 // 블록 할당 함수를 호출하여, 첫 번째 데이터 블록 인덱스를 할당 받음
301 int blk_index = alloc_block(&fs->sb, fs->block_map);
302 if (blk_index < 0)
303 return -ENOSPC; // 사용 가능한 블록 공간 부족 시 에러 반환
304
305 // 실제 블록 번호는 데이터 블록의 시작 위치에서 blk_index만큼 더한 값
306 uint32_t phys_block = fs->sb.data_block_start + blk_index;
307
308 // inode가 관리하는 직접(direct) 블록 포인터 중 첫 번째 항목을 설정
309 root_inode.direct[0] = phys_block;
310 root_inode.size =
311 SFUSE_BLOCK_SIZE; // 루트 디렉터리 크기를 블록 크기만큼 설정
312
313 // 데이터 블록 내에 루트 디렉터리의 기본 엔트리("." 및 "..")를 생성하기 위한
314 // 초기화
315 uint8_t block[SFUSE_BLOCK_SIZE] = {0}; // 블록 크기만큼 0으로 초기화
316 struct sfuse_dirent *entries =
317 (struct sfuse_dirent *)block; // 디렉터리 엔트리 구조체로 사용
318
319 // "."(현재 디렉터리)는 루트 inode를 가리킴
320 entries[0].ino = root;
321 strcpy(entries[0].name, ".");
322
323 // ".."(상위 디렉터리) 역시 루트 inode를 가리킴 (루트 디렉터리는 자신을
324 // 가리킴)
325 entries[1].ino = root;
326 strcpy(entries[1].name, "..");
327
328 // 구성된 디렉터리 데이터를 디스크의 실제 블록에 기록
329 if (write_block(backing_fd, phys_block, block) < 0)
330 return -EIO; // 블록 기록 실패 시 입출력 에러 반환
331
332 // 구성된 루트 inode를 디스크에 저장하여 동기화
333 if (inode_sync(backing_fd, &fs->sb, root, &root_inode) < 0)
334 return -EIO; // inode 동기화 실패 시 입출력 에러 반환
335
336 // ---- 모든 메타데이터의 최종 상태를 디스크에 동기화 ----
337
338 // 블록 비트맵과 inode 비트맵의 최종 상태를 디스크에 저장
339 bitmap_sync(backing_fd, fs->sb.block_bitmap_start, fs->block_map,
340 bmap_bytes);
341 bitmap_sync(backing_fd, fs->sb.inode_bitmap_start, fs->inode_map,
342 imap_bytes);
343
344 // 슈퍼블록의 최종 상태를 디스크에 저장하여 파일 시스템 초기화 완료
345 sb_sync(backing_fd, &fs->sb);
346
347 } else {
348 // 슈퍼블록이 이미 존재(정상 로드 성공) 시, 기존 메타데이터를 디스크에서
349 // 로드
350
351 size_t bmap_bytes = fs->sb.blocks_count / 8;
352 size_t imap_bytes = fs->sb.inodes_count / 8;
353
354 fs->block_map = calloc(1, bmap_bytes);
355 fs->inode_map = calloc(1, imap_bytes);
356
357 // 기존 비트맵 데이터를 디스크에서 메모리로 로드하여 파일 시스템 재구성
358 bitmap_load(backing_fd, fs->sb.block_bitmap_start, fs->block_map,
359 bmap_bytes);
360 bitmap_load(backing_fd, fs->sb.inode_bitmap_start, fs->inode_map,
361 imap_bytes);
362 }
363
364 // 초기화 과정이 모두 정상적으로 완료되었으므로 성공(0)을 반환
365 return 0;
366}

377void fs_destroy(void *private_data) {
378 // 전달받은 private_data 포인터를 struct sfuse_fs 타입으로 변환
379 struct sfuse_fs *fs = private_data;
380
381 // 비트맵 데이터를 디스크에 동기화하기 위한 메모리 크기 계산
382 // 블록 비트맵 크기: 전체 블록 수(blocks_count)를 비트맵 표현을 위해 바이트로
383 // 환산
384 size_t bmap_bytes = fs->sb.blocks_count / 8;
385
386 // inode 비트맵 크기: 전체 inode 수(inodes_count)를 비트맵 표현을 위해
387 // 바이트로 환산
388 size_t imap_bytes = fs->sb.inodes_count / 8;

392 // 반영된다.
393 bitmap_sync(fs->backing_fd, fs->sb.block_bitmap_start, fs->block_map,
394 bmap_bytes);
395
396 // 메모리에 존재하는 inode 비트맵 상태를 디스크에 동기화한다.
397 // 파일 시스템 종료 시 inode의 사용 여부를 디스크에 저장한다.
398 bitmap_sync(fs->backing_fd, fs->sb.inode_bitmap_start, fs->inode_map,
399 imap_bytes);
400
401 // 슈퍼블록 상태를 디스크에 동기화하여 최신의 파일 시스템 메타데이터를
402 // 유지한다.
403 sb_sync(fs->backing_fd, &fs->sb);
404
405 // 파일 시스템의 종료 작업 이후 메모리에 할당된 비트맵 메모리를 해제하여,
406 // 메모리 누수를 방지한다.
407 free(fs->block_map); // 블록 비트맵 메모리 해제
408 free(fs->inode_map); // inode 비트맵 메모리 해제

426int fs_resolve_path(struct sfuse_fs *fs, const char *path, uint32_t *ino_out) {
427 // 시작 지점을 루트 디렉터리의 inode로 설정
428 uint32_t cur = SFUSE_ROOT_INO;
429
430 // 예외 처리: 빈 문자열 또는 루트 디렉터리('/')가 들어오면 바로 루트 inode
431 // 반환
432 if (!path || path[0] == '\0' || strcmp(path, "/") == 0) {
433 *ino_out = cur;
434 return 0;
435 }
436
437 // 경로 문자열을 수정 가능하도록 복제(메모리 복사)
438 char *dup = strdup(path);
439 if (!dup)
440 return -ENOMEM; // 메모리 부족 시 에러 반환
441
442 // 경로 문자열을 "/" 기준으로 토큰(token)으로 나눈다.
443 // 예시: "/home/user/file" → ["home", "user", "file"]
444 char *token = strtok(dup, "/");
445
446 // 모든 토큰에 대해 순차적으로 디렉터리를 탐색
447 while (token) {
448 // 디렉터리 데이터를 메모리에 로드하기 위한 버퍼 크기를 계산한다.
449 // 해당 버퍼는 디렉터리 엔트리 정보를 저장하는 데 사용된다.
450 // SFUSE_NDIR_BLOCKS: 디렉터리 inode가 직접 참조할 수 있는 블록의 최대 개수.
451 // 직접 참조 블록 포인터 수 (12개)
452 // SFUSE_BLOCK_SIZE: 각 블록의 크기(바이트 단위)()
453 //
454 // 따라서 디렉터리 데이터를 로드할 수 있는 전체 버퍼 크기는 다음과 같다:
455 //
456 // buf_size = (직접 참조 블록 개수) × (각 블록 크기)
457 //
458 // 수식으로 표현하면:
459 // buf_size = SFUSE_NDIR_BLOCKS * SFUSE_BLOCK_SIZE;
460 //
461 // 예시:
462 // 만약 SFUSE_NDIR_BLOCKS가 12이고 SFUSE_BLOCK_SIZE가 1024 바이트라면,
463 // buf_size = 12 × 1024 = 12288 바이트(12KB)가 된다.
464 size_t buf_size = SFUSE_NDIR_BLOCKS * SFUSE_BLOCK_SIZE;
465 void *buf = malloc(buf_size);
466 if (!buf) {
467 free(dup);
468 return -ENOMEM; // 메모리 부족 시 에러 반환
469 }
470
471 // 현재 디렉터리의 모든 엔트리(entry)를 디스크에서 메모리로 로드한다.
472 // dir_load는 cur(현재 inode 번호)에 해당하는 디렉터리 데이터를 버퍼에
473 // 채운다.
474 if (dir_load(fs->backing_fd, &fs->sb, cur, buf) < 0) {
475 free(buf);
476 free(dup);
477 return -ENOENT; // 디렉터리를 읽을 수 없는 경우 존재하지 않음 에러 반환
478 }
479
480 // 버퍼의 데이터를 디렉터리 엔트리 배열로 캐스팅하여 접근 가능하도록 설정
481 struct sfuse_dirent *ents = (struct sfuse_dirent *)buf;
482
483 // 디렉터리 엔트리의 최대 개수를 계산
484 size_t max_entries = buf_size / sizeof(*ents);
485
486 // 다음 디렉터리 또는 파일의 inode 번호를 찾기 위한 임시 변수
487 uint32_t next_ino = 0;
488
489 // 현재 디렉터리 내의 엔트리 중에서 이름이 토큰과 일치하는 엔트리를 찾는다.
490 for (size_t i = 0; i < max_entries; i++) {
491 // ents[i].ino가 0이면 빈 엔트리이므로 무시
492 // 토큰과 같은 이름의 엔트리를 찾으면 next_ino에 inode 번호 저장 후 반복문
493 // 종료
494 if (ents[i].ino != 0 && strcmp(ents[i].name, token) == 0) {
495 next_ino = ents[i].ino;
496 break;
497 }
498 }
499
500 // 버퍼 사용이 끝났으므로 메모리 해제
501 free(buf);
502
503 // 해당하는 이름을 가진 엔트리를 찾지 못한 경우, 경로가 존재하지 않음을 의미
504 if (next_ino == 0) {
505 free(dup);
506 return -ENOENT; // 경로 존재하지 않음 에러 반환
507 }
508
509 // 다음 토큰 탐색을 위해 현재 inode 번호를 방금 찾은 inode 번호로 업데이트
510 cur = next_ino;
511
512 // 다음 토큰으로 넘어감
513 token = strtok(NULL, "/");
514 }
515
516 // 모든 토큰 탐색이 완료되면, 최종 inode 번호를 결과로 반환
517 free(dup); // 경로 문자열 복제본의 메모리 해제
518 *ino_out = cur; // 최종 inode 번호 저장
519 return 0; // 성공적으로 경로를 해석함
520}

543char *fs_split_path(const char *fullpath, uint32_t *parent_ino) {
544 char *dup = strdup(fullpath); // 경로를 수정하기 위해 복제본 생성
545 if (!dup)
546 return NULL; // 메모리 부족 시 실패 처리
547
548 char *slash = strrchr(dup, '/'); // 마지막 '/' 문자를 찾아 경로 분리 지점 설정
549 char *name;
550
551 if (!slash) {
552 // '/' 문자가 없을 경우 (예: "file.txt")
553 *parent_ino = SFUSE_ROOT_INO; // 부모는 루트 디렉터리로 설정
554 name = dup; // 전체 경로가 곧 파일명
555 } else if (slash == dup) {
556 // 경로가 루트에 바로 붙은 경우 (예: "/file.txt")
557 *parent_ino = SFUSE_ROOT_INO; // 부모는 루트 디렉터리
558 name = strdup(slash + 1); // '/' 다음부터 이름 복사 ("file.txt")
559 free(dup); // 최초 복제한 문자열은 더 이상 필요 없음
560 } else {
561 // 그 외 일반적인 경우 (예: "/home/user/file.txt")
562 *slash =
563 '\0'; // '/' 위치에서 문자열을 종료시켜 부모 경로만 남김 ("/home/user")
564
565 // 남은 부모 경로(dup)를 이용하여 부모 디렉터리의 inode 번호를 찾는다.
566 if (fs_resolve_path(get_fs_context(), dup, parent_ino) < 0) {
567 free(dup); // 부모 경로 탐색 실패 시 메모리 해제
568 return NULL; // 실패 반환
569 }
570
571 // '/' 뒤의 문자열("file.txt")을 이름으로 복사하여 반환
572 name = strdup(slash + 1);
573 free(dup); // 부모 경로 탐색 완료 후 메모리 해제
574 }
575
576 return name; // 호출한 곳에서 반드시 free(name) 수행 필요
577}
