113char *fs_split_path(const char *fullpath, uint32_t *parent_ino) {
114 char *dup = strdup(fullpath);
115 if (!dup)
116 return NULL;
117
118 char *slash = strrchr(dup, '/');
119 char *name;
120
121 if (!slash) {
122 *parent_ino = SFUSE_ROOT_INO;
123 name = dup;
124 } else if (slash == dup) {
125 *parent_ino = SFUSE_ROOT_INO;
126 name = strdup(slash + 1);
127 free(dup);
128 } else {
129 *slash = '\0';
130 if (fs_resolve_path(get_fs_context(), dup, parent_ino) < 0) {
131 free(dup);
132 return NULL;
133 }
134 name = strdup(slash + 1);
135 free(dup);
136 }
137
138 return name; // 호출한 곳에서 반드시 free(name)

142int fs_resolve_path(struct sfuse_fs *fs, const char *path, uint32_t *ino_out) {
143 uint32_t cur = SFUSE_ROOT_INO;
144 if (!path || path[0] == '\0' || strcmp(path, "/") == 0) {
145 *ino_out = cur;
146 return 0;
147 }
148 // 경로를 '/' 구분자로 토큰 분리
149 char *dup = strdup(path);
150 if (!dup)
151 return -ENOMEM;
152 char *token = strtok(dup, "/");
153 while (token) {
154 // 현재 디렉터리의 모든 엔트리 로드
155 size_t buf_size = SFUSE_NDIR_BLOCKS * SFUSE_BLOCK_SIZE;
156 void *buf = malloc(buf_size);
157 if (!buf) {
158 free(dup);
159 return -ENOMEM;
160 }
161 if (dir_load(fs->backing_fd, &fs->sb, cur, buf) < 0) {
162 free(buf);
163 free(dup);
164 return -ENOENT;
165 }
166 struct sfuse_dirent *ents = (struct sfuse_dirent *)buf;
167 size_t max_entries = buf_size / sizeof(*ents);
168 uint32_t next_ino = 0;
169 for (size_t i = 0; i < max_entries; i++) {
170 if (ents[i].ino != 0 && strcmp(ents[i].name, token) == 0) {
171 next_ino = ents[i].ino;
172 break;
173 }
174 }
175 free(buf);
176 if (next_ino == 0) {
177 free(dup);
178 return -ENOENT;
179 }
180 cur = next_ino;
181 token = strtok(NULL, "/");
182 }
183 free(dup);
184 *ino_out = cur;
185 return 0;
186}
