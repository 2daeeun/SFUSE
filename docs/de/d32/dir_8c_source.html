91 uint8_t *inode_map, struct sfuse_super *supermap) {
92 // 부모 디렉터리의 아이노드를 로드하여 inode 구조체에 저장한다.
93 struct sfuse_inode inode;
94 int res = inode_load(fd, sb, ino, &inode);
95 if (res < 0)
96 return res; // 아이노드 로딩에 실패하면 즉시 종료한다.
97
98 // 디렉터리 데이터 전체를 저장할 메모리 버퍼를 할당한다.
99 size_t buf_size = SFUSE_NDIR_BLOCKS * SFUSE_BLOCK_SIZE;
100 void *buf = malloc(buf_size);
101 if (!buf)
102 return -ENOMEM; // 메모리 할당 실패 시 오류 코드 반환
103
104 // 디스크로부터 디렉터리의 모든 직접 블록 데이터를 읽어온다.
105 res = dir_load(fd, sb, ino, buf);
106 if (res < 0) {
107 free(buf);
108 return res; // 블록 로드 실패 시 메모리 해제 후 종료한다.
109 }
110
111 // 버퍼를 디렉터리 엔트리 배열로 해석하여 최대 엔트리 개수 산정
112 struct sfuse_dirent *ents = (struct sfuse_dirent *)buf;
113 uint32_t max_entries = buf_size / sizeof(*ents);
114
115 // 디렉터리 내의 빈 슬롯을 탐색하여 새 엔트리를 추가한다.
116 for (uint32_t i = 0; i < max_entries; i++) {
117 if (ents[i].ino == 0) { // 빈 슬롯 발견 (아이노드 번호가 0이면 빈 슬롯)
118 ents[i].ino = child_ino; // 새 아이노드 번호를 할당
119 strncpy(ents[i].name, name, sizeof(ents[i].name) - 1); // 엔트리 이름 복사
120 ents[i].name[sizeof(ents[i].name) - 1] = '\0'; // 문자열의 NULL 종단 보장
121
122 // 수정된 전체 디렉터리 데이터를 디스크의 직접 블록에 다시 기록한다.
123 for (uint32_t b = 0; b < SFUSE_NDIR_BLOCKS; b++) {
124 if (inode.direct[b] == 0)
125 continue; // 사용되지 않은 블록은 건너뛴다.
126
127 res = write_block(fd, inode.direct[b],
128 (char *)buf + b * SFUSE_BLOCK_SIZE);
129 if (res < 0) {
130 free(buf);
131 return res; // 블록 기록 실패 시 메모리 해제 후 종료한다.
132 }
133 }
134
135 // 변경된 슈퍼블록과 블록 및 아이노드 비트맵을 디스크에 동기화한다.
136 sb_sync(fd, supermap);
137 bitmap_sync(fd, sb->block_bitmap_start, block_map, sb->blocks_count / 8);
138 bitmap_sync(fd, sb->inode_bitmap_start, inode_map, sb->inodes_count / 8);
139
140 free(buf); // 작업이 완료되었으므로 메모리 해제
141 return 0; // 성공적으로 엔트리 추가 완료
142 }
143 }
144
145 // 디렉터리 내 모든 슬롯이 사용 중이어서 추가 불가 (공간 부족)
146 free(buf);
147 return -ENOSPC; // 추가 공간 부족 오류 반환
148}

162int dir_remove_entry(int fd, const struct sfuse_super *sb, uint32_t ino,
163 const char *name) {
164 // 디렉터리 아이노드 로드
165 struct sfuse_inode inode;
166 int res = inode_load(fd, sb, ino, &inode);
167 if (res < 0)
168 return res;
169
170 // 디렉터리 데이터 블록 메모리에 로드
171 size_t buf_size = SFUSE_NDIR_BLOCKS * SFUSE_BLOCK_SIZE;
172 void *buf = malloc(buf_size);
173 if (!buf)
174 return -ENOMEM;
175
176 res = dir_load(fd, sb, ino, buf);
177 if (res < 0) {
178 free(buf);
179 return res;
180 }
181
182 struct sfuse_dirent *ents = (struct sfuse_dirent *)buf;
183 uint32_t max_entries = buf_size / sizeof(*ents);
184
185 // 이름이 일치하는 엔트리 탐색 및 삭제
186 for (uint32_t i = 0; i < max_entries; i++) {
187 if (ents[i].ino != 0 && strcmp(ents[i].name, name) == 0) {
188 ents[i].ino = 0;
189 ents[i].name[0] = '\0';
190
191 // 수정된 디렉터리 데이터를 디스크에 기록
192 for (uint32_t b = 0; b < SFUSE_NDIR_BLOCKS; b++) {
193 if (inode.direct[b] == 0)
194 continue;
195
196 res = write_block(fd, inode.direct[b],
197 (char *)buf + b * SFUSE_BLOCK_SIZE);
198 if (res < 0) {
199 free(buf);
200 return res;
201 }
202 }
203
204 free(buf);
205 return 0;
206 }
207 }
208
209 // 해당 이름의 엔트리가 없을 경우 메모리 해제 후 오류 반환
210 free(buf);
211 return -ENOENT;
212}
