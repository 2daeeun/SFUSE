47void fs_init_inode(const struct sfuse_super *sb, uint32_t ino, mode_t mode,
48 uid_t uid, gid_t gid, struct sfuse_inode *inode) {
49 (void)sb; // 미사용 파라미터로 인한 컴파일러 경고 방지
50 (void)ino; // 미사용 파라미터로 인한 컴파일러 경고 방지
51
52 // inode 메모리를 0으로 초기화하여 깨끗한 상태로 시작
53 memset(inode, 0, sizeof(*inode));
54
55 // 파일 타입과 접근 권한(mode)을 설정
56 inode->mode = mode;
57
58 // 아이노드 소유자의 사용자(uid)와 그룹(gid) ID를 설정
59 inode->uid = uid;
60 inode->gid = gid;
61
62 // 새로 생성된 파일/디렉터리이므로 크기를 0으로 설정
63 inode->size = 0;
64
65 // 접근(atime), 수정(mtime), 상태 변경(ctime) 시간을 현재 시각으로 초기화
66 inode->atime = inode->mtime = inode->ctime = (uint32_t)time(NULL);
67
68 // Direct 블록 포인터를 모두 0으로 초기화하여 데이터 블록을 아직 가리키지 않음
69 for (int i = 0; i < SFUSE_NDIR_BLOCKS; i++)
70 inode->direct[i] = 0;
71
72 // Single indirect, Double indirect 블록 포인터도 데이터가 없음을 나타내기
73 // 위해 0 설정
74 inode->indirect = 0;
75 inode->double_indirect = 0;
76
77 // 디렉터리면 기본 링크 수를 2로, 파일이면 1로 설정
78 inode->links = (S_ISDIR(mode) ? 2 : 1);
79}

113int inode_load(int fd, const struct sfuse_super *sb, uint32_t ino,
114 struct sfuse_inode *inode) {
115 /* [1단계] 아이노드 번호(ino) 유효성 검사 */
116 if (ino == 0 || ino >= sb->inodes_count)
117 return -EINVAL; // ino가 유효하지 않음 (0이거나 최대 범위 초과)
118
119 /* [2단계] 디스크에서 아이노드가 저장된 위치(offset) 계산 */
120 off_t off = ((off_t)sb->inode_table_start * SFUSE_BLOCK_SIZE) +
121 (ino * sizeof(*inode));
122 /*
123 * inode_table_start 블록 번호를 실제 바이트 단위로 변환하기 위해 블록
124 * 크기(SFUSE_BLOCK_SIZE)를 곱함. 거기에 inode 번호(ino)에 inode
125 * 크기(sizeof(*inode))를 곱하여 정확한 위치 계산.
126 */
127
128 /* [3단계] 디스크에서 아이노드 데이터를 읽어 메모리에 로드 */
129 ssize_t ret = disk_read(fd, inode, sizeof(*inode), off);
130 if (ret < 0)
131 return (int)ret; // 디스크 읽기 오류 발생 시 해당 오류 코드 반환
132
133 // 읽은 데이터 크기가 아이노드 크기와 불일치 시 입출력 오류 반환
134 if ((size_t)ret != sizeof(*inode))
135 return -EIO;
136
137 return 0; // 성공적으로 아이노드 로드 완료
138}

173int inode_sync(int fd, const struct sfuse_super *sb, uint32_t ino,
174 const struct sfuse_inode *inode) {
175 /* [1단계] 아이노드 번호(ino)의 유효성 검사 */
176 if (ino == 0 || ino >= sb->inodes_count)
177 return -EINVAL; // 아이노드 번호가 유효하지 않을 경우
178
179 /* [2단계] 아이노드 데이터를 기록할 디스크의 정확한 위치(offset) 계산 */
180 off_t off = ((off_t)sb->inode_table_start * SFUSE_BLOCK_SIZE) +
181 (ino * sizeof(*inode));
182 /*
183 * inode_table_start는 블록 번호를 나타내므로 바이트 단위로 변환하기 위해
184 * SFUSE_BLOCK_SIZE를 곱한다.
185 * 아이노드 번호(ino)에 아이노드 크기를 곱해 정확한 바이트 오프셋을 결정한다.
186 */
187
188 /* [3단계] 메모리의 아이노드를 디스크의 해당 위치에 기록 */
189 ssize_t ret = disk_write(fd, inode, sizeof(*inode), off);
190 if (ret < 0)
191 return (int)ret; // 디스크 기록 오류가 발생했을 때 오류 코드 반환
192
193 if ((size_t)ret != sizeof(*inode))
194 return -EIO; // 기록된 데이터의 크기가 아이노드 크기와 다르면 입출력 오류
195 // 반환
196
197 return 0; // 성공적으로 아이노드를 디스크에 기록함
198}

235int logical_to_physical(int fd, const struct sfuse_super *sb,
236 const struct sfuse_inode *inode, uint32_t lbn,
237 void *buf, uint32_t *pbn_out) {
238
239 (void)sb; // 미사용 파라미터로 인한 컴파일러 경고 방지
240
241 /* --- [1단계: Direct 블록 확인] --- */
242 if (lbn < SFUSE_NDIR_BLOCKS) {
243 // 요청한 논리 블록 번호가 Direct 블록의 범위 내에 있음
244 *pbn_out = inode->direct[lbn]; // 직접적인 배열 참조로 즉시 얻을 수 있음
245 return 0;
246 }
247
248 /* --- [2단계: Single Indirect 블록 확인] --- */
249 lbn -= SFUSE_NDIR_BLOCKS; // Direct 블록의 범위를 초과했으므로, Single
250 // indirect 범위로 보정
251
252 uint32_t entries = SFUSE_BLOCK_SIZE / sizeof(uint32_t);
253 // 하나의 블록에 저장 가능한 블록 주소의 개수를 계산
254 // (예: 4096 바이트 블록 / 4 바이트 주소 = 1024개의 블록 주소 저장 가능)
255
256 if (lbn < entries) {
257 // Single indirect 블록에서 찾을 수 있는 범위 내에 있음
258 if (inode->indirect == 0)
259 return -ENOENT; // Single indirect 블록 자체가 아직 할당되지 않은 상태
260
261 // indirect 블록 번호가 존재하면, 해당 블록을 디스크에서 읽는다
262 if (read_block(fd, inode->indirect, buf) < 0)
263 return -EIO; // indirect 블록을 읽는 도중 오류가 발생하면 입출력 오류 반환
264
265 // indirect 블록에서 블록 주소 배열을 가져옴
266 uint32_t *ptrs = (uint32_t *)buf;
267 *pbn_out = ptrs[lbn]; // indirect 블록의 주소 배열에서 요청한 논리 블록의
268 // 물리 블록 번호를 얻음
269 return 0;
270 }
271
272 /* --- [3단계: Double Indirect 블록 확인] --- */
273 lbn -= entries; // Single indirect 블록의 범위를 초과하였으므로, Double
274 // indirect 블록을 처리하기 위해 보정
275
276 uint32_t idx1 = lbn / entries; // Double indirect 첫 번째 레벨의 인덱스 계산
277 uint32_t idx2 = lbn % entries; // Double indirect 두 번째 레벨의 인덱스 계산
278
279 if (inode->double_indirect == 0)
280 return -ENOENT; // Double indirect 블록 자체가 할당되지 않은 상태
281
282 // double indirect의 첫 번째 레벨 블록을 읽는다.
283 if (read_block(fd, inode->double_indirect, buf) < 0)
284 return -EIO; // 첫 번째 레벨 블록을 읽을 때 오류 발생 시 입출력 오류 반환
285
286 uint32_t *ptrs1 = (uint32_t *)buf;
287 uint32_t block2 = ptrs1[idx1]; // 두 번째 레벨 블록 번호를 얻음
288
289 if (block2 == 0)
290 return -ENOENT; // 두 번째 레벨 블록이 할당되지 않은 경우
291
292 // 두 번째 레벨 블록을 디스크에서 읽는다.
293 if (read_block(fd, block2, buf) < 0)
294 return -EIO; // 두 번째 레벨 블록을 읽을 때 오류 발생 시 입출력 오류 반환
295
296 uint32_t *ptrs2 = (uint32_t *)buf;
297 *pbn_out = ptrs2[idx2]; // 최종적으로 변환된 물리 블록 번호를 얻음
298
299 return 0; // 성공적으로 물리 블록 번호 변환 완료
300}
