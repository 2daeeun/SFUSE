35int bitmap_load(int fd, uint32_t block_no, uint8_t *map, size_t map_size) {
36 ssize_t ret;
37
38 // 디스크에서 비트맵 데이터를 읽어 메모리(map)로 로드한다.
39 ret = disk_read(fd, map, map_size, (off_t)block_no * SFUSE_BLOCK_SIZE);
40 if (ret < 0)
41 return (int)ret; // disk_read 실패: 읽기 자체가 실패함
42
43 // 읽은 데이터 크기가 요청한 크기와 일치하는지 검증한다.
44 if ((size_t)ret != map_size)
45 return -EIO; // 읽은 데이터 크기가 예상 크기와 불일치
46
47 return 0; // 정상 종료
48}

99int alloc_block(struct sfuse_super *sb, uint8_t *block_map) {
100 // 데이터 블록 영역에서 사용 가능한 총 블록 개수를 계산
101 uint32_t total = sb->blocks_count - sb->data_block_start;
102
103 // 비트맵을 처음부터 끝까지 탐색하며 빈 블록을 찾음
104 for (uint32_t i = 0; i < total; i++) {
105 uint32_t byte_idx = i / 8; // 비트맵의 바이트 단위 인덱스
106 uint32_t bit_idx = i % 8; // 바이트 내의 비트 위치 (0~7)
107
108 // 현재 블록(i)이 사용 가능한 상태인지 확인 (비트 값이 0이면 사용 가능)
109 if (!(block_map[byte_idx] & (1 << bit_idx))) {
110 // 사용 가능한 블록 발견 시 해당 비트를 '1'로 설정하여 사용 중으로 표시
111 block_map[byte_idx] |= (1 << bit_idx);
112
113 // 슈퍼블록에서 사용 가능한 블록 개수 1 감소
114 sb->free_blocks--;
115
116 // 할당된 블록의 오프셋 반환 (0부터 시작)
117 return (int)i;
118 }
119 }
120
121 // 모든 블록이 이미 사용 중이면 ENOSPC(공간 부족 오류) 반환
122 return -ENOSPC;
123}

143 uint32_t byte_idx = offset / 8; // 비트맵 내의 바이트 단위 인덱스 계산
144 uint32_t bit_idx = offset % 8; // 바이트 내에서의 비트 위치 계산 (0~7)
145
146 // 비트맵에서 해당 블록의 비트를 '0'으로 설정 (빈 상태로 표시)
147 block_map[byte_idx] &= ~(1 << bit_idx);
148
149 // 슈퍼블록 내 빈 블록 개수를 1 증가 (가용 블록 수 갱신)
150 sb->free_blocks++;

172int alloc_inode(struct sfuse_super *sb, uint8_t *inode_map) {
173 // 아이노드 0은 예약이므로 1부터 탐색을 시작
174 for (uint32_t i = 1; i < sb->inodes_count; i++) {
175 uint32_t byte_idx = i / 8; // 비트맵 내의 바이트 인덱스 계산
176 uint32_t bit_idx = i % 8; // 바이트 내의 비트 위치 계산 (0~7)
177
178 // 현재 아이노드(i)가 빈 상태인지 검사 (0이면 빈 상태)
179 if (!(inode_map[byte_idx] & (1 << bit_idx))) {
180 // 사용 가능한 아이노드를 발견하면, 비트맵에 사용 중으로 표시 (비트=1)
181 inode_map[byte_idx] |= (1 << bit_idx);
182
183 // 슈퍼블록의 남은 빈 아이노드 수를 1 감소시킴
184 sb->free_inodes--;
185
186 // 할당된 아이노드 번호를 반환 (번호는 1부터 시작)
187 return (int)i;
188 }
189 }
190
191 // 빈 아이노드를 찾지 못했을 때 공간 부족(-ENOSPC) 오류 반환
192 return -ENOSPC;
193}

214 // 아이노드 번호 유효성 검사 (0번 아이노드 및 유효 범위 초과 시 무시)

223
224 // 슈퍼블록의 빈 아이노드 개수를 1 증가시킴

226}
